package tekstikayttis;

public class KonsoliIO implements IO{
    private Scanner lukija = new Scanner(System.in);

    @Override
    public String nextLine() {
        return lukija.nextLine();
    }

    @Override
    public void print(String m) {
        System.out.println(m);
    }
}
